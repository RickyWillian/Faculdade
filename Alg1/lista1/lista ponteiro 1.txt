#include <stdio.h>
#include <stdlib.h>


int main(int argc, char *argv[]) {
	
	int a = 10, *pa;
	double b= 5.0, *pb;
	char c = 'v', *pc;
	
	pa= &a;
	pc= &c;
	pb= &b;
	
	printf("\nA = %d", a);
	printf("\nB = %f", b);
	printf("\nC = %c", c);
	
	*pa = 50;
	*pb = 5.5;
	*pc = 'f';
	
	printf("\nA = %d", a);
	printf("\nB = %f", b);
	printf("\nC = %c", c);
	
	return 0;
}
