#include <stdio.h>

int main() {
    double salario = 5000.00; // Salário inicial em 2000
    double aumento = 0.015;   // Aumento de 1,5% em 2021
    int ano;

    // Aplicar aumento de 1,5% em 2021
    for (ano = 2000; ano <= 2021; ano++) {
        if (ano == 2021) {
            salario += salario * aumento;
        }
    }

    // Aplicar aumentos a partir de 2022
    for (ano = 2022; ano <= 2024; ano++) {
        aumento *= 2; // Dobrar o percentual de aumento a cada ano
        salario += salario * aumento;
    }

    printf("O salário atual do funcionário em 2024 é: R$%.2f\n", salario);

    return 0;
}

#include <stdio.h>

int main() {
    int aluno, r = 0, e = 0, a = 0;
    float notaa, notab, media, mediac = 0;

    for (aluno = 0; aluno < 6; aluno++) {
        printf("\nDigite a nota a do aluno %d: ", aluno + 1);
        scanf("%f", &notaa); 
        printf("Digite a nota b do aluno %d: ", aluno + 1);
        scanf("%f", &notab);

        media = (notaa + notab) / 2;
        mediac += media;

        printf("A media do aluno eh: %.2f\n", media); 
        if (media <= 3) {
            printf("Reprovado\n");
            r++;
        } else if (media > 3.0 && media <= 7.0) {
            printf("Exame\n");
            e++;
        } else {
            printf("Aprovado\n");
            a++;
        }
    }

    mediac /= 6;
    printf("\nA media da sala eh: %.2f\n", mediac); 
    printf("A quantidade de aprovados eh: %d\n", a);
    printf("A quantidade de exame eh: %d\n", e);
    printf("A quantidade de reprovados eh: %d\n", r);

    return 0;
}
